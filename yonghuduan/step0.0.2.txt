已有基础组件系统 (cloudTipModal)

类目和标签选择界面

exports.main = async (event, context) => {
  const { categoryId, tags } = event

  try {
    // 1. AI文案生成 (调用DeepSeek API)
    const prompt = `你是一个电商买家，根据以下标签：${tags.join('、')}，写一段60字左右的中文评价，语气自然，多使用Emoji，体现真实购物体验。`

    // 2. 内容安全审核
    const isSafe = await checkContentSafety(reviewText)
    const finalText = isSafe ? reviewText : getDefaultText()

    // 3. 匹配图片
    const imgList = await getRandomImages(categoryId, 3)

    return {
      success: true,
      data: {
        text: finalText,
        imgList: imgList
      }
    }
  } catch (error) {
    return {
      success: false,
      error: {
        code: error.code || 'GENERATE_FAILED',
        message: error.message
      }
    }
  }
}

2.2 创建 createLotteryCode 云函数
文件位置: cloudfunctions/createLotteryCode/

index.js 核心逻辑:


const cloud = require('wx-server-sdk')

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
})

const db = cloud.database()
const _ = db.command

exports.main = async (event, context) => {
  const { openid } = event

  try {
    // 生成唯一6位随机码
    let code = generateRandomCode()
    let attempts = 0
    const maxAttempts = 10

    // 确保码唯一
    while (attempts < maxAttempts) {
      const existing = await db.collection('lottery_codes')
        .where({ code: code })
        .count()

      if (existing.total === 0) break

      code = generateRandomCode()
      attempts++
    }

    // 插入数据库
    const result = await db.collection('lottery_codes').add({
      data: {
        code: code,
        status: 0,
        openid: openid,
        createTime: new Date()
      }
    })

    return {
      success: true,
      data: { code: code }
    }
  } catch (error) {
    return {
      success: false,
      error: {
        code: 'CODE_GENERATE_FAILED',
        message: error.message
      }
    }
  }
}

function generateRandomCode() {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789'
  let code = ''
  for (let i = 0; i < 6; i++) {
    code += chars.charAt(Math.floor(Math.random() * chars.length))
  }
  return code
}

index.js 核心逻辑:


const app = getApp()

Page({
  data: {
    categories: [],        // 类目列表
    categoryIndex: -1,     // 选中的类目索引
    currentTags: [],       // 当前类目的标签
    selectedTags: [],      // 已选择的标签
    canGenerate: false,    // 是否可以生成
    loading: false
  },

  onLoad() {
    this.loadCategories()
  },

  // 加载类目列表
  async loadCategories() {
    try {
      const res = await wx.cloud.database().collection('products').get()
      this.setData({
        categories: res.data
      })
    } catch (error) {
      wx.showToast({ title: '加载类目失败', icon: 'none' })
    }
  },

  // 选择类目
  onCategoryChange(e) {
    const index = e.detail.value
    this.setData({
      categoryIndex: index,
      currentTags: this.data.categories[index].tags,
      selectedTags: [],
      canGenerate: false
    })
  },

  // 选择/取消标签
  onTagToggle(e) {
    const tag = e.currentTarget.dataset.tag
    let selectedTags = [...this.data.selectedTags]

    const idx = selectedTags.indexOf(tag)
    if (idx > -1) {
      selectedTags.splice(idx, 1)
    } else {
      selectedTags.push(tag)
    }

    const categoryId = this.data.categories[this.data.categoryIndex]._id
    this.setData({
      selectedTags,
      canGenerate: selectedTags.length > 0
    })
  },

  // 生成评价
  async onGenerate() {
    if (!this.data.canGenerate) return

    this.setData({ loading: true })

    try {
      const categoryId = this.data.categories[this.data.categoryIndex]._id

      const res = await wx.cloud.callFunction({
        name: 'generateReview',
        data: {
          categoryId: categoryId,
          tags: this.data.selectedTags
        }
      })

      if (res.result.success) {
        // 跳转到结果页
        wx.navigateTo({
          url: `/pages/result/result?data=${encodeURIComponent(JSON.stringify(res.result.data))}`
        })
      } else {
        wx.showToast({
          title: res.result.error.message || '生成失败',
          icon: 'none'
        })
      }
    } catch (error) {
      wx.showToast({ title: '网络错误，请重试', icon: 'none' })
    } finally {
      this.setData({ loading: false })
    }
  }
})

index.js 核心逻辑:


Page({
  data: {
    reviewText: '',
    imageUrls: [],
    saveCount: 0,
    rewardCode: '',
    showModal: false,
    customerServiceQR: '/images/customer-service-qrcode.png'
  },

  onLoad(options) {
    const data = JSON.parse(decodeURIComponent(options.data))
    this.setData({
      reviewText: data.text,
      imageUrls: data.imgList
    })
  },

  // 复制文案
  onCopyText() {
    wx.setClipboardData({
      data: this.data.reviewText,
      success: () => {
        wx.showToast({ title: '文案已复制' })
      }
    })
  },

  // 预览图片
  onPreviewImage(e) {
    const index = e.currentTarget.dataset.index
    wx.previewImage({
      current: this.data.imageUrls[index],
      urls: this.data.imageUrls
    })
  },

  // 保存单张图片
  async onSaveSingle(e) {
    const index = e.currentTarget.dataset.index
    const url = this.data.imageUrls[index]

    try {
      await this.saveImageToAlbum(url)
      wx.showToast({ title: '保存成功', icon: 'success' })
    } catch (error) {
      this.handlePermissionError()
    }
  },

  // 保存所有图片
  async onSaveAll() {
    wx.showLoading({ title: '保存中...' })

    try {
      let successCount = 0

      for (const url of this.data.imageUrls) {
        try {
          await this.saveImageToAlbum(url)
          successCount++
        } catch (error) {
          console.error('保存失败:', error)
        }
      }

      wx.hideLoading()

      if (successCount === this.data.imageUrls.length) {
        wx.showToast({ title: '全部保存成功', icon: 'success' })
        await this.getRewardCode()
      } else {
        wx.showToast({
          title: `成功保存${successCount}张`,
          icon: 'none'
        })
      }
    } catch (error) {
      wx.hideLoading()
      wx.showToast({ title: '保存失败', icon: 'none' })
    }
  },

  // 保存图片到相册
  saveImageToAlbum(url) {
    return new Promise((resolve, reject) => {
      wx.downloadFile({
        url: url,
        success: (res) => {
          wx.saveImageToPhotosAlbum({
            filePath: res.tempFilePath,
            success: resolve,
            fail: reject
          })
        },
        fail: reject
      })
    })
  },

  // 处理权限错误
  handlePermissionError() {
    wx.getSetting({
      success: (res) => {
        if (res.authSetting['scope.writePhotosAlbum'] === false) {
          wx.showModal({
            title: '需要相册权限',
            content: '保存图片需要您授权访问相册',
            confirmText: '去设置',
            success: (modalRes) => {
              if (modalRes.confirm) {
                wx.openSetting()
              }
            }
          })
        }
      }
    })
  },

  // 获取奖励码
  async getRewardCode() {
    try {
      const app = getApp()
      const res = await wx.cloud.callFunction({
        name: 'createLotteryCode',
        data: {
          openid: app.globalData.openid || wx.getStorageSync('openid')
        }
      })

      if (res.result.success) {
        this.setData({
          rewardCode: res.result.data.code,
          showModal: true
        })
      }
    } catch (error) {
      console.error('获取奖励码失败:', error)
    }
  },

  // 复制奖励码
  onCopyCode() {
    wx.setClipboardData({
      data: this.data.rewardCode,
      success: () => {
        wx.showToast({ title: '验证码已复制' })
      }
    })
  },

  // 关闭弹窗
  onCloseModal() {
    this.setData({ showModal: false })
  }
})
