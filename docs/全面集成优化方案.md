# 好评宝系统全面集成优化方案

## 📋 项目概述

**优化目标**: 打通小程序用户端、网页商家管理端、后端和数据库之间的数据流和业务逻辑整合

**当前问题**:
- 数据源不统一（小程序云开发 vs SQLAlchemy SQL）
- 认证机制不完整（模拟微信登录）
- API不一致（不同端口、响应格式）
- 缺少实时同步机制

**优化范围**:
- 微信小程序用户端
- 网页商家管理端
- Flask后端API
- SQLite数据库
- 系统间集成通信

```python
from app import db

class User(db.Model):

    __tablename__ = 'merchants'

    name = db.Column(db.String(100))
    contact = db.Column(db.String(50))
    phone = db.Column(db.String(20))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

    # 关联关系
    users = db.relationship('User', backref='merchant', lazy=True)
    categories = db.relationship('Category', backref='merchant', lazy=True)

    __tablename__ = 'categories'

    id = db.Column(db.String(64), primary_key=True)
    name = db.Column(db.String(50))
    merchant_id = db.Column(db.String(64), db.ForeignKey('merchants.id'))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

    # 关联关系
    tags = db.relationship('Tag', backref='category', lazy=True)
    evaluations = db.relationship('Evaluation', backref='category', lazy=True)

    """评价标签"""
    __tablename__ = 'tags'

    id = db.Column(db.String(64), primary_key=True)
    name = db.Column(db.String(50))
    category_id = db.Column(db.String(64), db.ForeignKey('categories.id'))
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

class Evaluation(db.Model):
    """评价记录"""
    __tablename__ = 'evaluations'

    id = db.Column(db.String(64), primary_key=True)
    content = db.Column(db.Text)
    user_id = db.Column(db.String(64), db.ForeignKey('users.id'))
    category_id = db.Column(db.String(64), db.ForeignKey('categories.id'))
    generated_at = db.Column(db.DateTime, default=datetime.utcnow)

class LotteryRecord(db.Model):

class Prize(db.Model):
    """奖品"""

    id = db.Column(db.String(64), primary_key=True)
    name = db.Column(db.String(100))

class CodePool(db.Model):
```

```python
# app/auth/wechat_oauth.py
import requests
from flask import current_app

class WeChatOAuth:
    @staticmethod
    def get_openid(code):
        """通过code获取微信openid"""
        url = f"https://api.weixin.qq.com/sns/jscode2session"
        params = {
            'appid': current_app.config['WECHAT_APPID'],
            'secret': current_app.config['WECHAT_SECRET'],
            'js_code': code,
            'grant_type': 'authorization_code'
        }
        response = requests.get(url, params=params)
        data = response.json()
        return data.get('openid')

    @staticmethod
    def get_user_info(access_token, openid):
        """获取微信用户信息"""
        url = "https://api.weixin.qq.com/sns/userinfo"
        params = {
            'access_token': access_token,
            'openid': openid
        }
        return requests.get(url, params=params).json()
```

#### 2.2 Redis缓存层

**缓存策略**:
```python
# app/cache/redis_cache.py
import redis
from functools import wraps

redis_client = redis.Redis(host='localhost', port=6379, db=0)

def cache_data(key_prefix, expire_time=3600):
    """缓存装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            cache_key = f"{key_prefix}:{hash(str(args) + str(kwargs))}"

            # 尝试从缓存获取
            cached_data = redis_client.get(cache_key)
            if cached_data:
                return json.loads(cached_data)

            # 缓存不存在，执行函数
            result = func(*args, **kwargs)
            redis_client.setex(cache_key, expire_time, json.dumps(result))
            return result
        return wrapper
    return decorator
```

**缓存使用场景**:
- 用户会话缓存
- 类目标签数据缓存
- 系统配置缓存

```python
# app/permissions/rbac.py
from enum import Enum

class Permission(Enum):
    EVALUATION_CREATE = 'evaluation:create'
    EVALUATION_READ = 'evaluation:read'
    EVALUATION_DELETE = 'evaluation:delete'
    LOTTERY_MANAGE = 'lottery:manage'
    USER_MANAGE = 'user:manage'
    SYSTEM_CONFIG = 'system:config'

class Role:
    MERCHANT_ADMIN = ['*']  # 全部权限
    MERCHANT_USER = [
        Permission.EVALUATION_CREATE.value,
        Permission.EVALUATION_READ.value,
        Permission.LOTTERY_MANAGE.value
    ]

def require_permission(permission):
    """权限验证装饰器"""
    def decorator(f):
        @wraps(f)
        def decorated_function(*args, **kwargs):
            user_permissions = get_user_permissions(current_user.id)
            if permission not in user_permissions and '*' not in user_permissions:
                return {'message': 'Permission denied'}, 403
            return f(*args, **kwargs)
        return decorated_function
    return decorator
```

```python
from contextlib import contextmanager
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

@contextmanager
def db_transaction():
    """数据库事务上下文管理器"""
    session = Session()
    try:
        yield session
        session.commit()
    except Exception as e:
        session.rollback()
        raise e
    finally:
        session.close()
```

```python
# app/utils/consistency_checker.py
def check_data_consistency():
    """检查数据一致性"""
    # 检查用户-商家关联
    orphaned_users = User.query.filter(
        User.merchant_id.notin_(
            Merchant.query.with_entities(Merchant.id)
        )
    ).all()

    # 检查评价-类目关联
    orphaned_evaluations = Evaluation.query.filter(
        Evaluation.category_id.notin_(
            Category.query.with_entities(Category.id)
        )
    ).all()

    return {
        'orphaned_users': len(orphaned_users),
        'orphaned_evaluations': len(orphaned_evaluations)
    }
```

```python
class BusinessMetrics:
    def __init__(self):
        self.daily_counts = defaultdict(int)
        self.success_rates = defaultdict(list)

    def record_evaluation(self, success):
        self.daily_counts['evaluation'] += 1
        self.success_rates['evaluation'].append(success)

        return {
            'evaluations_count': self.daily_counts['evaluation'],
```
